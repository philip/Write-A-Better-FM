\usepackage{psboxit,alltt,fancybox,epsfig}
\newcommand{\n}{\ensuremath{\backslash{}}n}
% This environment is for example code (ebox = example box)
% Usage: \begin{ebox} <code> \end{ebox}
% The one caveat is that { and } must be escaped.
\newenvironment{ebox}%
 {\begin{boxitpara}{box 0.90 setgray fill}\begin{alltt}%
    \catcode`\$=12 \catcode`\#=12 \catcode`\%=12}
 {\catcode`\%=\active \catcode`\#=\active \catcode`\$=\active%
    \end{alltt} \end{boxitpara}}
